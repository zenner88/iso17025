CREATE TABLE jenisuji (
	ID INTEGER UNSIGNED NOT NULL DEFAULT 31 PRIMARY KEY,
	NAMA TINYTEXT,
	NAMA2 TINYTEXT,
	JMLVAR INT UNSIGNED NOT NULL,
	RUMUS TINYTEXT
);

CREATE TABLE varjenisuji (
	IDJENISUJI INTEGER UNSIGNED NOT NULL ,
	VAR CHAR(5) NOT NULL ,
	NAMA TINYTEXT,
	MANUAL SMALLINT UNSIGNED NOT NULL,
	RUMUS TINYTEXT,
	PRIMARY KEY(IDJENISUJI,VAR)
);
